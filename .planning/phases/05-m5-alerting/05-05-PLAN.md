#[tokio::test]
async fn test_alert_routing_e2e() {
    // Setup: Create mock sinks that record calls
    let mut webhook_calls = Arc::new(Mutex::new(Vec::new()));
    let mut slack_calls = Arc::new(Mutex::new(Vec::new()));
    let mut pagerduty_calls = Arc::new(Mutex::new(Vec::new()));
    
    // Create AlertRouter with test config (low thresholds for testing)
    let config = AlertConfig {
        min_severity: Severity::Warning,
        dedup_window_secs: 60,
        rate_limit_per_min: 1000,
    };
    let router = AlertRouter::new(config);
    
    // Mock sinks: Instead of real HTTP, record calls in Arc<Mutex<Vec>>
    // (Or use a real httptest::server for the webhook)
    
    // Create test findings: 1 CRIT, 1 WARN, 1 INFO
    let crit_finding = Finding::new(..., "crit-detector", "/", "Critical Issue", Severity::Critical, ...);
    let warn_finding = Finding::new(..., "warn-detector", "/", "Warning Issue", Severity::Warning, ...);
    let info_finding = Finding::new(..., "info-detector", "/", "Info Issue", Severity::Info, ...);
    
    // Inject findings into router
    let start = Instant::now();
    router.route(&crit_finding).await.unwrap();
    router.route(&warn_finding).await.unwrap();
    router.route(&info_finding).await.unwrap();  // May be skipped by severity gate
    let elapsed = start.elapsed();
    
    // Verify
    assert!(elapsed < Duration::from_secs(5), "Alert routing took >5s");
    assert_eq!(webhook_calls.lock().await.len(), 2);  // CRIT and WARN
    assert_eq!(slack_calls.lock().await.len(), 2);
    assert_eq!(pagerduty_calls.lock().await.len(), 2);
}
```

async fn test_crit_alert_delivery_within_5s_sla() {
    let router = AlertRouter::new(AlertConfig { min_severity: Severity::Critical, ... });
    
    // Use httptest::Server or mock that measures call time
    let webhook_server = httptest::Server::run();
    let webhook_sink = WebhookSink::new(webhook_server.url("/alerts"));
    router = router.with_sink(Box::new(webhook_sink));
    
    let crit_finding = Finding::new(..., Severity::Critical, ...);
    
    let start = Instant::now();
    router.route(&crit_finding).await.unwrap();
    let elapsed = start.elapsed();
    
    // Verify latency and webhook received POST
    assert!(elapsed < Duration::from_secs(5), "Took {}ms", elapsed.as_millis());
    assert_eq!(webhook_server.requests().len(), 1);
    
    let request = &webhook_server.requests()[0];
    assert_eq!(request.method, "POST");
    assert_eq!(request.path, "/alerts");
}
```
