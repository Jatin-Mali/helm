    2. After updating findings and sparklines, add:
       ```rust
       // Spawn parallel fleet collection (non-blocking)
       if !self.fleet_collection_in_flight {
           self.fleet_collection_in_flight = true;
           let registry = self.remote_registry.clone();
           tokio::task::spawn(async move {
               let results = registry.parallel_collect().await;
               // Aggregate results into FleetHostStatus
               let fleet_hosts = results.into_iter().map(|(host_id, result)| {
                   match result {
                       Ok(findings) => {
                           let crit = findings.iter().filter(|f| f.severity == "CRIT").count();
                           let warn = findings.iter().filter(|f| f.severity == "WARN").count();
                           let info = findings.iter().filter(|f| f.severity == "INFO").count();
                           FleetHostStatus {
                               host_id,
                               name: findings.first().map(|f| f.host_name.clone()).unwrap_or_default(),
                               reachable: true,
                               crit, warn, info,
                               last_refresh: Some(Instant::now()),
                               error: None,
                           }
                       },
                       Err(e) => FleetHostStatus {
                           host_id,
                           name: "unknown".to_string(),
                           reachable: false,
                           crit: 0, warn: 0, info: 0,
                           last_refresh: Some(Instant::now()),
                           error: Some(e.to_string()),
                       }
                   }
               }).collect();
               // Store results in self.dashboard.data.fleet_hosts (via Arc<Mutex> or similar)
           });
       }
       ```
    3. Add flag to DashboardData: `fleet_collection_in_flight: bool` (initialized to false).
    4. Reset flag when collection completes (via Arc<Mutex<bool>> or message channel).
  </action>
