       - Arrow Down: increment fleet_selected_row (clamp to fleet_hosts.len() - 1)
       - Enter: switch active_remote to the selected fleet_host.host_id, trigger dashboard refresh
    3. Example:
       ```rust
       if self.current_tab == OpsTab::Fleet {
           match key.code {
               KeyCode::Up => {
                   if let Some(row) = self.fleet_selected_row {
                       self.fleet_selected_row = Some(row.saturating_sub(1));
                   } else {
                       self.fleet_selected_row = Some(0);
                   }
               },
               KeyCode::Down => {
                   let row = self.fleet_selected_row.unwrap_or(0);
                   let max = self.dashboard.data.fleet_hosts.len().saturating_sub(1);
                   self.fleet_selected_row = Some((row + 1).min(max));
               },
               KeyCode::Enter => {
                   if let Some(row) = self.fleet_selected_row {
                       if let Some(host) = self.dashboard.data.fleet_hosts.get(row) {
                           self.active_remote = Some(host.host_id);
                           // Trigger dashboard refresh
                       }
                   }
               },
               _ => {},
           }
       }
       ```
